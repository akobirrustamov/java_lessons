File Upload — Spring Boot'da

spring:
  servlet:
    multipart:
      enabled: true
      max-file-size: 10MB
      max-request-size: 20MB
      file-size-threshold: 2KB        # disk'ga yozish chegarasi
@Service
@RequiredArgsConstructor
public class FileService {
    private static final Set<String> ALLOWED_TYPES =
        Set.of("image/jpeg", "image/png", "image/webp", "application/pdf");
    private static final long MAX_SIZE = 10 * 1024 * 1024;  // 10MB

    public StoredFile store(MultipartFile file) {
        // 1. Validation
        if (file.isEmpty()) throw new BadRequestException("File is empty");
        if (file.getSize() > MAX_SIZE) throw new BadRequestException("Too large");

        String contentType = file.getContentType();
        if (!ALLOWED_TYPES.contains(contentType)) {
            throw new BadRequestException("Type not allowed: " + contentType);
        }

        // 2. Magic byte check (faqat content-type'ga ishonmang!)
        try (InputStream is = file.getInputStream()) {
            String detected = new Tika().detect(is);
            if (!ALLOWED_TYPES.contains(detected)) {
                throw new BadRequestException("Detected type: " + detected);
            }
        }

        // 3. Unique fayl nomi (orig nomi xavfli — path traversal)
        String ext = StringUtils.getFilenameExtension(file.getOriginalFilename());
        String filename = UUID.randomUUID() + "." + ext;

        // 4. Save (S3 yoki lokal)
        return storage.save(filename, file.getInputStream());
    }
}

Avatar upload endpoint

POST /users/{id}/avatar — rasm yuklash. Faqat JPG/PNG, max 2MB. 

@PostMapping(value = "/users/{id}/avatar",
             consumes = MediaType.MULTIPART_FORM_DATA_VALUE)
public Map<String, String> uploadAvatar(
        @PathVariable Long id,
        @RequestParam("file") MultipartFile file) {

    if (file.isEmpty()) throw new BadRequestException("Empty");
    if (file.getSize() > 2 * 1024 * 1024) {
        throw new BadRequestException("Max 2MB");
    }

    String ct = file.getContentType();
    if (!Set.of("image/jpeg", "image/png").contains(ct)) {
        throw new BadRequestException("Only JPG/PNG");
    }

    String url = fileService.uploadAvatar(id, file);
    return Map.of("url", url);
}

@GetMapping("/files/{id}")
public ResponseEntity<Resource> download(@PathVariable Long id) {
    StoredFile file = fileService.findById(id);
    Resource resource = fileService.loadAsResource(file.getPath());

    return ResponseEntity.ok()
        .contentType(MediaType.parseMediaType(file.getContentType()))
        .header(HttpHeaders.CONTENT_DISPOSITION,
                "attachment; filename=\"" + file.getOriginalName() + "\"")
        .contentLength(file.getSize())
        .body(resource);
}

// Streaming uchun (katta fayllar)
@GetMapping("/files/{id}/stream")
public ResponseEntity<StreamingResponseBody> stream(@PathVariable Long id) {
    StoredFile file = fileService.findById(id);

    StreamingResponseBody body = output -> {
        try (InputStream is = storage.openStream(file.getPath())) {
            is.transferTo(output);
        }
    };

    return ResponseEntity.ok()
        .contentType(MediaType.parseMediaType(file.getContentType()))
        .body(body);
}

// pom.xml

@Service
public class ImageProcessor {

    public Map<String, byte[]> createThumbnails(MultipartFile file) throws IOException {
        BufferedImage original = ImageIO.read(file.getInputStream());

        Map<String, byte[]> result = new HashMap<>();
        result.put("thumbnail", resize(original, 50, 50));
        result.put("medium",    resize(original, 200, 200));
        result.put("large",     resize(original, 500, 500));
        return result;
    }

    private byte[] resize(BufferedImage img, int w, int h) throws IOException {
        ByteArrayOutputStream out = new ByteArrayOutputStream();
        Thumbnails.of(img)
            .size(w, h)
            .outputFormat("jpg")
            .outputQuality(0.85)
            .toOutputStream(out);
        return out.toByteArray();
    }
}

S3 + Presigned URL (direct upload)

Katta fayllar uchun: client'ga vaqtinchalik signed URL bering, u to'g'ridan-to'g'ri S3'ga yuklasin. Server proxy bo'lib turmaydi. 

@RestController
@RequestMapping("/files")
@RequiredArgsConstructor
public class FileController {
    private final S3Presigner s3Presigner;

    @PostMapping("/upload-url")
    public Map<String, String> getUploadUrl(
            @RequestBody UploadRequest req) {

        String key = "uploads/" + UUID.randomUUID() + "-" + req.getFilename();

        PutObjectRequest putRequest = PutObjectRequest.builder()
            .bucket("my-bucket")
            .key(key)
            .contentType(req.getContentType())
            .build();

        PutObjectPresignRequest presignRequest = PutObjectPresignRequest.builder()
            .signatureDuration(Duration.ofMinutes(15))
            .putObjectRequest(putRequest)
            .build();

        URL url = s3Presigner.presignPutObject(presignRequest).url();

        return Map.of("uploadUrl", url.toString(), "key", key);
    }

    @PostMapping("/confirm")
    public StoredFile confirmUpload(@RequestBody ConfirmRequest req) {
        // Client S3'ga yuklab bo'lgan — endi DB'ga metadata yozamiz
        return fileService.confirmUploaded(req.getKey(), req.getMetadata());
    }
}

Email Sending — nazariya

Email — sinxron yuborish XAVFLI. SMTP server sekin (1-5 sekund), uzilishi mumkin. User HTTP javobini kutib qoladi.

Yechim — async yoki queue.

1️⃣ @Async — oddiy, in-process

2️⃣ RabbitMQ/Kafka — alohida service, retry, idempotency

3️⃣ 3rd-party (SendGrid, Mailgun) — REST API, deliverability

SINXRON (YOMON):
User → POST /register → save user → sendEmail() → SMTP → response (5s)

ASYNC (YAXSHI):
User → POST /register → save user → queue.publish(event) → response (50ms)
                                          ↓
                          Background worker → SMTP

YANGI:
                  ┌─ retry 3 marta
Email worker ────┼─ failure notification
                  └─ exponential backoff

        } catch (Exception e) {
            log.error("Failed to send email to {}", user.getEmail(), e);
        }
        return CompletableFuture.completedFuture(null);

Email with Retry + Outbox

Email yuborishda xatolik bo'lishi mumkin. Outbox pattern bilan reliability: DB'ga yozamiz → background worker o'qib yuboradi. 

@Entity
public class EmailOutbox {
    @Id @GeneratedValue private Long id;
    private String to;
    private String subject;
    @Column(columnDefinition = "TEXT")
    private String htmlBody;
    private EmailStatus status;     // PENDING, SENT, FAILED
    private int attempts;
    private LocalDateTime nextAttemptAt;
    private LocalDateTime createdAt;
}

@Service
@RequiredArgsConstructor
public class OutboxEmailService {
    private final EmailOutboxRepository repo;
    private final JavaMailSender mailSender;

    @Transactional
    public void enqueue(String to, String subject, String html) {
        EmailOutbox email = new EmailOutbox();
        email.setTo(to); email.setSubject(subject); email.setHtmlBody(html);
        email.setStatus(EmailStatus.PENDING);
        email.setNextAttemptAt(LocalDateTime.now());
        repo.save(email);
    }

    @Scheduled(fixedDelay = 30_000)  // har 30 sekund
    @Transactional
    public void processOutbox() {
        List<EmailOutbox> pending = repo.findPendingForSending(LocalDateTime.now());
        for (EmailOutbox email : pending) {
            try {
                sendNow(email);
                email.setStatus(EmailStatus.SENT);
            } catch (Exception e) {
                email.setAttempts(email.getAttempts() + 1);
                if (email.getAttempts() >= 5) {
                    email.setStatus(EmailStatus.FAILED);
                } else {
                    // Exponential backoff: 2^n daqiqa
                    email.setNextAttemptAt(LocalDateTime.now()
                        .plusMinutes((long) Math.pow(2, email.getAttempts())));
                }
            }
            repo.save(email);
        }
    }
}

Caching — nazariya va patternlar

Cache nima uchun? Sekin operatsiyalar natijasini xotirada saqlash. DB so'rovi 100ms → Cache 1ms.

Asosiy patternlar:

PatternTavsif
Cache-Aside (Lazy)App o'zi tekshiradi: cache → DB → cache'ga yozish. Eng ko'p ishlatiladi.
Write-ThroughYozishda DB va Cache birga yangilanadi. Slow write, consistent.
Write-Behind (Write-Back)Cache'ga yoziladi, DB'ga async. Tez, lekin xavfli (data loss).
Read-ThroughCache miss'da cache o'zi DB'dan oladi. Cloud servislarda ko'p.
Cache invalidation — kompyuter fanidagi eng qiyin 2 ta narsadan biri (Phil Karlton). Eski ma'lumotni qaytarish va to'g'ri vaqtda o'chirish — qiyin masala.

Redis-based Rate Limiter

Foydalanuvchi daqiqada 100 ta so'rovdan oshib ketmasin. Redis bilan sliding window. 

@Service
@RequiredArgsConstructor
public class RateLimiter {
    private final StringRedisTemplate redis;
    private static final int LIMIT = 100;
    private static final Duration WINDOW = Duration.ofMinutes(1);

    public boolean tryAcquire(String userId) {
        String key = "rate:" + userId + ":" + (System.currentTimeMillis() / 60_000);

        Long count = redis.opsForValue().increment(key);
        if (count == 1) {
            redis.expire(key, WINDOW);
        }

        return count <= LIMIT;
    }
}

@Component
@RequiredArgsConstructor
public class RateLimitFilter extends OncePerRequestFilter {
    private final RateLimiter rateLimiter;

    @Override
    protected void doFilterInternal(HttpServletRequest req,
                                   HttpServletResponse res,
                                   FilterChain chain) throws IOException, ServletException {
        String userId = extractUserId(req);
        if (!rateLimiter.tryAcquire(userId)) {
            res.setStatus(HttpStatus.TOO_MANY_REQUESTS.value());
            res.getWriter().write("{\"error\":\"Rate limit exceeded\"}");
            return;
        }
        chain.doFilter(req, res);
    }
}

Distributed Lock with Redis

Bir vaqtda faqat 1 ta instance kritik operatsiyani bajarsin (masalan: nightly job). 

@Service
@RequiredArgsConstructor
public class DistributedLockService {
    private final StringRedisTemplate redis;

    /**
     * Redis'da atomik SET NX (faqat bo'lmasa o'rnatish).
     * Lock owner — UUID, expire — qulflar abadiy qolmasligi uchun.
     */
    public boolean tryLock(String key, String owner, Duration ttl) {
        Boolean acquired = redis.opsForValue()
            .setIfAbsent("lock:" + key, owner, ttl);
        return Boolean.TRUE.equals(acquired);
    }

    public void releaseLock(String key, String owner) {
        // Lua script — atomik tekshirish va o'chirish
        String script = """
            if redis.call('GET', KEYS[1]) == ARGV[1] then
                return redis.call('DEL', KEYS[1])
            else
                return 0
            end
        """;
        redis.execute(
            new DefaultRedisScript<>(script, Long.class),
            List.of("lock:" + key), owner
        );
    }
}

@Scheduled(cron = "0 0 2 * * *")  // har kun 02:00

    } finally {
        lockService.releaseLock("nightly-job", owner);
    }

@ExtendWith(MockitoExtension.class)
class UserServiceTest {

    @Mock UserRepository userRepo;
    @Mock PasswordEncoder encoder;
    @InjectMocks UserService service;

    @Test
    @DisplayName("Yangi user yaratganda parol BCrypt'ga aylanishi kerak")
    void shouldHashPasswordOnCreate() {
        // GIVEN — tayyorlash
        UserRequest req = new UserRequest("Akobir", "test@mail.com", "plainPass");
        when(encoder.encode("plainPass")).thenReturn("$2a$12$hashed...");
        when(userRepo.save(any(User.class))).thenAnswer(inv -> inv.getArgument(0));

        // WHEN — sinash
        User result = service.create(req);

        // THEN — tekshirish
        assertThat(result.getName()).isEqualTo("Akobir");
        assertThat(result.getPasswordHash()).startsWith("$2a$12$");

        // Verify — mock chaqirilganini tekshirish
        verify(encoder).encode("plainPass");
        verify(userRepo).save(argThat(u -> u.getEmail().equals("test@mail.com")));
    }

    @Test
    void shouldThrowWhenEmailExists() {
        when(userRepo.existsByEmail("test@mail.com")).thenReturn(true);

        assertThatThrownBy(() -> service.create(new UserRequest("X", "test@mail.com", "p")))
            .isInstanceOf(EmailExistsException.class)
            .hasMessageContaining("test@mail.com");
    }

    @ParameterizedTest
    @ValueSource(strings = {"", " ", "no-at-sign", "@no-domain"})
    void shouldRejectInvalidEmails(String email) {
        assertThatThrownBy(() -> service.validate(email))
            .isInstanceOf(InvalidEmailException.class);
    }
}

Integration test with Testcontainers

@SpringBootTest(webEnvironment = WebEnvironment.RANDOM_PORT)
@Testcontainers
@Transactional
class FullStackIntegrationTest {

    @Container
    static PostgreSQLContainer<?> postgres =
        new PostgreSQLContainer<>("postgres:16-alpine");

    @Container
    static GenericContainer<?> redis =
        new GenericContainer<>("redis:7-alpine").withExposedPorts(6379);

    @DynamicPropertySource
    static void config(DynamicPropertyRegistry registry) {
        registry.add("spring.datasource.url",      postgres::getJdbcUrl);
        registry.add("spring.datasource.username", postgres::getUsername);
        registry.add("spring.datasource.password", postgres::getPassword);
        registry.add("spring.data.redis.host",     redis::getHost);
        registry.add("spring.data.redis.port",     () -> redis.getMappedPort(6379));
    }

    @Autowired TestRestTemplate rest;
    @Autowired UserRepository userRepo;

    @Test
    void shouldRegisterAndLogin() {
        // Register
        ResponseEntity<UserResponse> register = rest.postForEntity(
            "/api/auth/register",
            new RegisterRequest("Akobir", "a@mail.com", "Pass123!"),
            UserResponse.class
        );
        assertThat(register.getStatusCode()).isEqualTo(HttpStatus.CREATED);
        assertThat(userRepo.findByEmail("a@mail.com")).isPresent();

        // Login
        ResponseEntity<TokenResponse> login = rest.postForEntity(
            "/api/auth/login",
            new LoginRequest("a@mail.com", "Pass123!"),
            TokenResponse.class
        );
        assertThat(login.getStatusCode()).isEqualTo(HttpStatus.OK);
        assertThat(login.getBody().getAccessToken()).isNotBlank();
    }
}

GitHub Actions — misol

name: CI/CD Pipeline

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  build-and-test:
    runs-on: ubuntu-latest

    services:
      postgres:
        image: postgres:16
        env:
          POSTGRES_PASSWORD: test
        ports: [5432:5432]
        options: --health-cmd pg_isready --health-interval 5s

    steps:
      - uses: actions/checkout@v4

      - name: Setup JDK 21
        uses: actions/setup-java@v4
        with:
          java-version: '21'
          distribution: 'temurin'
          cache: 'maven'

      - name: Build
        run: ./mvnw clean package -B

      - name: Test
        run: ./mvnw test -B
        env:
          DB_URL: jdbc:postgresql://localhost:5432/test

      - name: SonarCloud
        run: ./mvnw sonar:sonar -Dsonar.token=${{ secrets.SONAR_TOKEN }}

      - name: Build Docker
        if: github.ref == 'refs/heads/main'
        run: |
          docker build -t my-app:${{ github.sha }} .
          docker tag my-app:${{ github.sha }} my-app:latest

      - name: Push to registry
        if: github.ref == 'refs/heads/main'
        run: |
          echo "${{ secrets.DOCKER_PASSWORD }}" | docker login -u "${{ secrets.DOCKER_USERNAME }}" --password-stdin
          docker push my-app:${{ github.sha }}

      - name: Deploy
        if: github.ref == 'refs/heads/main'
        run: |
          ssh user@server "docker pull my-app:latest && \
                          docker compose up -d --no-deps app"
